Elimina un FileMultimediale dalla libreria.

Il metodo ricerca il FileMultimediale dato dopodichè crea un array temporaneo con tutti gli oggetti eccetto quello dato e sostituisce il vecchio array col nuovo.

Parametri:
fFileMultimediale da eliminare dalla libreria. 
  {
    if(indice == 0)
      return false;
    else
    {
      FileMultimediale tmp [] = new FileMultimediale[indice-1];
      int j=0;
      for(int i=0; i< indice ; i++)
      {
        if(!fm[i].Confronta(f))
        {
          tmp[j] = fm[i];
          j++;
        }
      }
      
      fm = tmp;
      indice --;
      return true;
